sub new {
my $type = shift;
my %params = @_;
my $self = {}; 


        bless $self, $type;
        $self->_init;
        #print "\n[", ref ($self) ,"]\n";;
    return $self;                 
}

sub _init {
my $self = shift;
$self->{'row'}->{'pre'} = ``'';
$self->{'row'}->{'post'} = ``'';
$self->{'taxa'}->{'pre'} = ``'';
$self->{'taxa'}->{'post'} = ``'';
$self->{'block'}->{'pre'} = ``'';
$self->{'block'}->{'post'} = ``'';
$self->{'bracketed_block'}->{'pre'} = ``'';
$self->{'bracketed_block'}->{'post'} = ``'';

        return 1;
}

sub row {
my $self = shift;

        return $self->{row}->{$pos};
}
